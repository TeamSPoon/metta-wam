;;;; Define DeBruijn Index
  (: DeBruijn Type)
  (: VarIdx 
    (-> Nat DeBruijn))
;;;; Map a DeBruijn Index to an given variable
  (: idx2var 
    (-> DeBruijn 
      (List Variable) Atom))
  (= 
    (idx2var 
      (VarIdx Z) 
      (Cons $head $tail)) $head)
  (= 
    (idx2var 
      (VarIdx (S $k)) 
      (Cons $head $tail)) 
    (idx2var 
      (VarIdx $k) $tail))
;;;; Map a DeBruijn Index in a given pattern to a variable
  (: Debruijn2var 
    (-> Atom 
      (List Variable) Atom))
  (= 
    (Debruijn2var 
      (VarIdx $k) $varlist) 
    (idx2var 
      (VarIdx $k) $varlist))
  (= 
    (Debruijn2var $symbol $varlist) 
    (if 
      (== 
        (get-type $symbol) %Undefined%) $symbol 
      (empty)))
  (= 
    (Debruijn2var 
      ($link $first $second) $varlist) 
    ($link 
      (Debruijn2var $first $varlist) 
      (Debruijn2var $second $varlist)))
