  (: Pr

;;;; Constants ;;
;;;;;;;;;;;;;;;;;

  (: prior-alpha 
    (-> Number))
  (=

    (-> Number))
  (=

  (: lookahead 
    (-> Number))
  (= 
    (lookahead) 1.0)

  (: max-count 
    (-> Number))
  (= 
    (max-count) 1000000000.0)

;;;; Methods ;;
;;;;;;;;;;;;;;;
;;;; Convert count to confidence using the formula
;;;;
;;;; confidence = count / (count + lookahead)
  (: count->confidence 
    (-> Number Number))
  (= 
    (count->confidence $cnt) 
    (/ $cnt 
      (+ $cnt 
        (lookahead))))
;;;; Convert confidence to count using the formula
;;;;
;;;; count = (confidence * lookahead) / (1 - confidence)
  (: confidence->count 
    (-> Number Number))
  (= 
    (confidence->count $conf) 
    (if 
      (approxEq 1.0 $conf 1.0e-9) 
      (max-count) 
      (/ 
        (* $conf 
          (lookahead)) 
        (- 1.0 $conf))))
;;;; Increment the negative count of a given truth value, by
;;;; incrementing its total count without incrementing the positive
;;;; count.
  (: inc-neg-count 
    (-> TruthValue TruthValue))
  (= 
    (inc-neg-count (STV $s $c)) 
    (let* 
      ( ($tot_cnt (confidence->count $c)) 
        ($pos_cnt (* $s $tot_cnt)) 
        ($new_tot_cnt (+ $tot_cnt 1))) 
      (STV 
        (/ $pos_cnt $new_tot_cnt) 
        (count->confidence $new_tot_cnt))))

;;;; incrementing its total count and its positive count.
  (: inc-pos-count 
    (-> TruthValue TruthValue))
  (= 
    (inc-pos-count (STV $s $c)) 
    (let* 
      ( ($tot_cnt (confidence->count $c)) 
        ($pos_cnt (* $s $tot_cnt)) 
        ($new_pos_cnt (+ $pos_cnt 1)) 
        ($new_tot_cnt (+ $tot_cnt 1)))

        (/ $new_pos_cnt $new_tot_cnt)

  (: count

    (count (Bl $))

;;;; Convert count to confidence using the formula
;;;;
;;;; confidence = count / (count + lookahead)
  (: count->confidence 
    (-> Number Number))
  (= 
    (count->confidence $cnt) 
    (/ $cnt 
      (+ $cnt 
        (lookahead))))

;;;;
;;;; count = (confidence * lookahead) / (1 - confidence)
  (: confidence->count 
    (-> Number Number))
  (= 
    (confidence->count $conf) 
    (if 
      (approxEq 1.0 $conf 1.0e-9) 
      (max-count) 
      (/ 
        (* $conf 
          (lookahead)) 
        (- 1.0 $conf))))
;;;; Increment the negative count of a given truth value, by
;;;; incrementing its total count without incrementing the positive
;;;; count.
  (: inc-neg-count 
    (-> TruthValue TruthValue))
  (= 
    (inc-neg-count (STV $s $c)) 
    (let* 
      ( ($tot_cnt (confidence->count $c)) 
        ($pos_cnt (* $s $tot_cnt)) 
        ($new_tot_cnt (+ $tot_cnt 1))) 
      (STV 
        (/ $pos_cnt $new_tot_cnt) 
        (count->confidence $new_tot_cnt))))

;;;; incrementing its total count and its positive count.
  (: inc-pos-count 
    (-> TruthValue TruthValue))
  (= 
    (inc-pos-count (STV $s $c)) 
    (let* 
      ( ($tot_cnt (confidence->count $c)) 
        ($pos_cnt (* $s $tot_cnt)) 
        ($new_pos_cnt (+ $pos_cnt 1)) 
        ($new_tot_cnt (+ $tot_cnt 1))) 
      (STV 
        (/ $new_pos_cnt $new_tot_cnt)

;;;; be supported at the moment).
  (: count 
    (-> TruthValue Number))

    (count (Bl $))

      (+ $cnt

  (: confidence->count 
    (-> Number Number))
  (= 
    (confidence->count $conf) 
    (if 
      (approxEq 1.0 $conf 1.0e-9) 
      (max-count)

          (lookahead))

;;;; incrementing its total count without incrementing the positive
;;;; count.
  (: inc-neg-count 
    (-> TruthValue TruthValue))
  (= 
    (inc-neg-count (STV $s $c)) 
    (let* 
      ( ($tot_cnt (confidence->count $c)) 
        ($pos_cnt (* $s $tot_cnt)) 
        ($new_tot_cnt (+ $tot_cnt 1))) 
      (STV 
        (/ $pos_cnt $new_tot_cnt)
